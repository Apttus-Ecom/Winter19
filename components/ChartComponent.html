projects/apttus/elements/src/lib/chart/chart.component.ts

Description

Chart component displays the data as a chart based on the configuration passed in to this component.

Usage

import { ChartModule } from '@apttus/elements';

@NgModule({
  imports: [ChartModule, ...]
})
export class AppModule {}

Implements

OnChanges OnInit OnDestroy

Example

// Basic Usage.
<apt-chart
  [data]="chartData"
  [type]="chartType"
></apt-chart>

// Chart with override settings.
<apt-chart
  [data]="chartData"
  [type]="chartType"
  [chartTitle]="myTitle"
  [colorPalette]="colors"
  [showLegend]="flag"
  [legendPosition]="'top'"
></apt-chart>
